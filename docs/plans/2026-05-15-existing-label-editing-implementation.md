# Existing Label Editing Implementation Plan

> **For agentic workers:** REQUIRED: Use `/skill:orchestrator-implements` (in-session, orchestrator implements), `/skill:subagent-driven-development` (in-session, subagents implement), or `/skill:executing-plans` (parallel session) to implement this plan. Steps use checkbox syntax for tracking.

**Goal:** Let users edit existing repository labels on PRs and issues through provider-aware APIs, cached catalog sync, desktop detail UI, and command palette action.

**Architecture:** Add provider-neutral label catalog/read and label mutation capabilities. Persist full repo label catalogs with catalog-membership metadata, refresh them through background sync with conditional provider fetches, and expose provider-aware API routes for cached catalog reads and full-set label replacement. Build a shared Svelte label picker driven by detail stores and a small label editor state store.

**Tech Stack:** Go, Huma, SQLite migrations, `go-github/v84`, GitLab Go SDK, Gitea SDK, Forgejo SDK, Svelte 5 runes, Vitest, Playwright, Bun, docker-compose provider fixtures.

```go
func (d *DB) ReplaceRepoLabelCatalog(ctx context.Context, repoID int64, labels []Label, syncedAt time.Time) error {
    syncedAt = syncedAt.UTC()
    return d.Tx(ctx, func(tx *sql.Tx) error {
        if _, err := tx.ExecContext(ctx, `UPDATE middleman_labels SET catalog_present = 0 WHERE repo_id = ?`, repoID); err != nil {
            return fmt.Errorf("clear label catalog: %w", err)
        }
        for i := range labels {
            labels[i].CatalogPresent = true
            labels[i].CatalogSeenAt = &syncedAt
            if labels[i].UpdatedAt.IsZero() {
                labels[i].UpdatedAt = syncedAt
            }
        }
        if _, err := upsertLabelsTx(ctx, tx, repoID, labels); err != nil {
            return err
        }
        if _, err := tx.ExecContext(ctx, `
            UPDATE middleman_repos
            SET label_catalog_synced_at = ?, label_catalog_checked_at = ?, label_catalog_sync_error = ''
            WHERE id = ?`, syncedAt, syncedAt, repoID); err != nil {
            return fmt.Errorf("mark label catalog synced: %w", err)
        }
        return nil
    })
}

func (d *DB) ListRepoLabelCatalog(ctx context.Context, repoID int64) ([]Label, LabelCatalogFreshness, error) {
    rows, err := d.ReadDB().QueryContext(ctx, `
        SELECT id, repo_id, COALESCE(platform_id, 0), platform_external_id, name, description, color,
               is_default, updated_at, catalog_present, catalog_seen_at
        FROM middleman_labels
        WHERE repo_id = ? AND catalog_present = 1
        ORDER BY lower(name), name`, repoID)
    if err != nil {
        return nil, LabelCatalogFreshness{}, fmt.Errorf("list repo label catalog: %w", err)
    }
    defer rows.Close()
    labels := []Label{}
    for rows.Next() {
        var label Label
        var seen sql.NullTime
        if err := rows.Scan(&label.ID, &label.RepoID, &label.PlatformID, &label.PlatformExternalID, &label.Name, &label.Description, &label.Color, &label.IsDefault, &label.UpdatedAt, &label.CatalogPresent, &seen); err != nil {
            return nil, LabelCatalogFreshness{}, fmt.Errorf("scan repo label catalog: %w", err)
        }
        if seen.Valid {
            label.CatalogSeenAt = &seen.Time
        }
        labels = append(labels, label)
    }
    if err := rows.Err(); err != nil {
        return nil, LabelCatalogFreshness{}, fmt.Errorf("iterate repo label catalog: %w", err)
    }
    freshness, err := d.GetRepoLabelCatalogFreshness(ctx, repoID)
    if err != nil {
        return nil, LabelCatalogFreshness{}, err
    }
    return labels, freshness, nil
}
```

func normalizeLabelNames(names []string) ([]string, error) {
    seen := map[string]struct{}{}
    out := make([]string, 0, len(names))
    for _, raw := range names {
        name := strings.TrimSpace(raw)
        if name == "" {
            return nil, fmt.Errorf("label name must not be empty")
        }
        if _, ok := seen[name]; ok {
            return nil, fmt.Errorf("duplicate label %q", name)
        }
        seen[name] = struct{}{}

async function toggle(label: IssueLabel): Promise<void> {
  if (pendingName !== null) return;
  const current = new Set(currentLabels.map((item) => item.name));
  if (current.has(label.name)) current.delete(label.name);
  else current.add(label.name);
  const nextNames = [...current];
  const previous = currentLabels;
  pendingName = label.name;
  error = null;
  try {
    const path = providerItemPath(kind === "pull" ? "pulls" : "issues", ref, "/labels");
    const { data, error: requestError } = await client.PUT(path, {
      params: { path: { ...providerRouteParams(ref), number } },
      body: { labels: nextNames },
    });
    if (requestError) throw new Error(requestError.detail ?? requestError.title ?? "failed to update labels");
    currentLabels = data?.labels ?? [];
    onlabelschange(currentLabels);
  } catch (err) {
    currentLabels = previous;
    error = err instanceof Error ? err.message : String(err);
  } finally {
    pendingName = null;
  }
}
